# 健身管理应用项目分析报告

## 1. 项目概述

这是一个基于全栈技术构建的健身管理应用，提供健身计划管理、饮食计划管理、AI生成个性化计划等功能，帮助用户科学管理健身和饮食。

### 3.2 服务器端API

**登录接口** (`server/api/login.post.js`)
```javascript
import getNeon from "../utils/neon";
import bcrypt from "bcrypt";

export default defineEventHandler(async (event) => {
  const { email, password } = await readBody(event);

  // 验证参数
  if (!email || !password) {
    return { success: false, message: "邮箱和密码不能为空" };
  }

  // 连接数据库
  const mySql = getNeon();

  try {
    // 查询用户信息
    const [userRows] = await mySql`SELECT id, username, email, password FROM gym_app_user WHERE email = ${email}`;

    if (!userRows || !userRows.id) {
      return { success: false, message: "用户不存在" };
    }

    // 验证密码
    const isPasswordValid = await bcrypt.compare(password, userRows.password);

    if (!isPasswordValid) {
      return { success: false, message: "密码错误" };
    }

    // 返回用户信息
    return {
      success: true,
      message: "登录成功",
      user: {
        id: userRows.id,
        username: userRows.username,
        email: userRows.email,
      },
    };
  } catch (error) {
    console.error("[Login] 数据库查询错误:", error);
    return { success: false, message: "登录失败，请稍后重试" };
  }
});
```

### 3.2 控制器层

**地点控制器** (`PlaceController.java`)
```java
package com.gym.place.controller;

import com.gym.place.exception.InvalidCoordinatesException;
import com.gym.place.service.PlaceService;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

import java.util.HashMap;
import java.util.Map;

/**
 * 地点控制器
 * 提供根据经纬度获取地点名称的API接口
 */
@RestController
@RequestMapping("/api")
public class PlaceController {
    
    private final PlaceService placeService;
    
    public PlaceController(PlaceService placeService) {
        this.placeService = placeService;
    }
    
    /**
     * 根据经纬度获取地点名称
     * @param lon 经度
     * @param lat 纬度
     * @return 包含地点名称的JSON响应
     */
    @GetMapping("/place")
    public Map<String, String> getPlaceName(
            @RequestParam(name = "lon", required = true) double lon,
            @RequestParam(name = "lat", required = true) double lat) {
        
        // 验证经纬度范围
        validateCoordinates(lon, lat);
        
        // 调用服务获取地点名称
        String placeName = placeService.getPlaceNameByCoordinates(lon, lat);
        
        // 构造返回结果
        Map<String, String> result = new HashMap<>();
        result.put("placeName", placeName);
        
        return result;
    }
    
    /**
     * 验证经纬度是否在合理范围内
     */
    private void validateCoordinates(double longitude, double latitude) {
        if (longitude < -180 || longitude > 180) {
            throw new InvalidCoordinatesException("经度必须在-180到180之间");
        }
        if (latitude < -90 || latitude > 90) {
            throw new InvalidCoordinatesException("纬度必须在-90到90之间");
        }
    }
}
```

### 3.3 服务层

**高德地图地点服务实现** (`AmapPlaceServiceImpl.java`)
```java
package com.gym.place.service.impl;

import com.gym.place.service.PlaceService;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.reactive.function.client.WebClient;

/**
 * 高德地图地点服务实现类
 * 核心功能：调用高德地图Web服务API完成「逆地理编码」（经纬度→具体地址）
 */
@Service
public class AmapPlaceServiceImpl implements PlaceService {

    private static final Logger logger = LoggerFactory.getLogger(AmapPlaceServiceImpl.class);

    private final String amapKey;
    private final String regeoApiPath;
    private final WebClient webClient;

    /**
     * 构造函数：注入配置 + 初始化WebClient
     */
    public AmapPlaceServiceImpl(
            @Value("${amap.key}") String amapKey,
            @Value("${amap.regeo-url-template}") String regeoApiPath) {
        this.amapKey = amapKey;
        this.regeoApiPath = regeoApiPath;
        validateAmapKey();
        this.webClient = initWebClient();
    }

    /**
     * 核心业务方法：根据经纬度获取格式化地址
     */
    @Override
    public String getPlaceNameByCoordinates(double longitude, double latitude) {
        // 验证经纬度合法性
        if (!isValidCoordinates(longitude, latitude)) {
            logger.warn("【经纬度校验失败】经纬度不合法");
            return "请求地址失败";
        }

        try {
            // 调用高德API
            Map<String, Object> apiResponse = callAmapRegeoApi(longitude, latitude);
            // 解析API响应
            String formattedAddress = parseRegeoResponse(apiResponse);
            
            return formattedAddress != null ? formattedAddress : "请求地址失败";
        } catch (Exception e) {
            logger.error("【地址解析异常】", e);
            return "请求地址失败";
        }
    }

    // 其他辅助方法...
}
```
